program
#include<stdio.h>
int fib(int num)
{
    if(num==0)
    return 0;
    else if(num==1||num==2)
    return 1;
    else
    return fib(num-1)+fib(num-2);
}
int main()
{
    int num=8;
    printf("%d",fib(num));
}

program - with recursion
#include<stdio.h>

int min_value(int arr[],int i,int j)// arr{},1,8
{
    int min_index=i-1;
    for(;i<=j;i++)
    {
       if(arr[min_index]>arr[i])
       {
           min_index=i;
       }
       
    }
    return min_index;
}

void sorting(int arr[],int start,int end)
{
  int min_index,temp;
  if(start==end)
  return ;
  min_index=min_value(arr,start+1,end);
  temp=arr[min_index];
  arr[min_index]=arr[start];
  arr[start]=temp;
  sorting(arr,start+1,end);
  
  
}
int main()
{
    int n,arr[100],itr;
    scanf("%d",&n);//9
    for(itr=0;itr<n;itr++)
    scanf("%d",&arr[itr]);
    sorting(arr,0,n-1);//arr{},0,8
    for(itr=0;itr<n;itr++)
    printf("%d ",arr[itr]);
}
