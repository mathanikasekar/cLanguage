#include <stdio.h>
#include <stdlib.h>
#include<limits.h>

int largest(int arr[],int i,int n)
{
    static int max=-98765432;
    if(i==n)
        return max;
    else
        if(max<arr[i])
        max=arr[i];
    largest(arr,i+1,n);

}
int main()
{
    int arr[]={2,3,4,6,8,20,45},n;
    n=sizeof(arr)/sizeof(int);
    printf("%d",largest(arr,0,n));
    return 0;
}
