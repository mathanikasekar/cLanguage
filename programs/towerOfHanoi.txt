#include <stdio.h>
#include <stdlib.h>
void TOH(int n,char source,char helping,char dest)
{
    if(n==1){
        printf("Move a disk from %c to %c\n",source, dest);
        return ;
    }
    TOH(n-1,source,dest,helping);
    printf("move a disk from %c to %c \n",source,dest);
    TOH(n-1,helping,source,dest);
}
int main()
{
    int n=7;
    char source='A',helping='B',dest='C';
    TOH(n,source,helping,dest);
    return 0;
}
