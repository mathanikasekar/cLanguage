#include<stdio.h>
#include<stdlib.h>
struct Node
{
int data;
struct Node *next;
};
struct Node* insert(struct Node *first,int n,int element)
{
    struct Node *p=first;     
    for(int i=0;i<n-1;i++)
    {
        p=p->next;
    }
    struct Node *new=(struct Node*)malloc(sizeof(struct Node));
    new->data=element;
    if(n==0)
    {
    new->next=first;
    first=new;
    }
    else
    {
        new->next=p->next;
        p->next=new;

    }
    return first;

}
void display(struct Node *head) //f(n)=n+n+n=3n->O(n)
{
    while(head!=NULL) //n unit
    {
        printf("%d ",head->data); //n unit
        head=head->next;  // n unit
    }
}
