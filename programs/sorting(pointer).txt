#include<stdio.h>
void swap(int *num1,int *num2)
{
    int temp;
    temp=*num1;
    *num1=*num2;
    *num2=temp;
}
void sorting(int arr[],int start,int end)
{
    int itr;
    if(start==end)
    return;
    for(itr=start;itr<end;itr++)
    if(arr[itr]>arr[itr+1])
    swap(&arr[itr],&arr[itr+1]);
    sorting(arr,start,end-1);
}

int main()
{
    int n,arr[100],itr;
    scanf("%d",&n);//9
    for(itr=0;itr<n;itr++)
    scanf("%d",&arr[itr]);
    sorting(arr,0,n-1);//arr{},0,8
    for(itr=0;itr<n;itr++)
    printf("%d ",arr[itr]);
}
