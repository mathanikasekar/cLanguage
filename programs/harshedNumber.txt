int sum_of_digit(int n)//111  11 1 0
{
    static int sum=0;
 if (n==0)
    return 0;
 return sum=n%10+sum_of_digit(n/10);
 /*1+sum_of_digit(11);1+2=3
 1+sum_of_digit(1);1+1=2
 1+sum_of_digit(0);1+0=1*/

}
int isharshed(int n)
{
    if(n%sum_of_digit(n)==0)
        return 1;
    else
        return 0;
}
