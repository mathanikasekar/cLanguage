struct Node* insert(struct Node *first,int n,int element)
{//f(n)=1+n-1+n-2+1+1+1+1+1=2n+3=O(n)
    struct Node *p=first;     // 1 unit
    for(int i=0;i<n-1;i++)   //n-1 unit
    {
        p=p->next;       //n-2 unit
    }
    struct Node *new=(struct Node*)malloc(sizeof(struct Node)); // 1 unit
    new->data=element;   // 1 unit
    if(n==0)     // 1 unit
    {
    new->next=first; // 1 unit
    first=new;   // 1 unit
    }
    else
    {
        new->next=p->next;
        p->next=new;

    }
    return first;   // 1 unit

}
