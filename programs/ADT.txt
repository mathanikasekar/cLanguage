#include<stdio.h>
struct array
{
   int *A;
   int n;
   int length;
};
void append(struct array *arr,int element)
{
    arr->A[arr->length]=element;
    arr->length++;
}
void display(struct array arr)
{

    int i;
    for(i=0;i<arr.length;i++)
    {

        printf("%d ",arr.A[i]);
    }
}
void insert(struct array *arr,int index,int element)
{

    int i;
    for(i=arr->length-1;i>=index;i--)
    {

        arr->A[i+1]=arr->A[i];
    }
   arr->A[index]=element;
    arr->length++;
}
void remmove(struct array *arr,int index)
{

    int i;
    for(i=index;i<=arr->length;i++)
    {
        arr->A[i]=arr->A[i+1];
    }
    arr->length--;
}
void set(struct array *arr,int index,int element)
{

   arr->A[index]=element;
}
void get(struct array arr,int index)
{
   int flag=0;
    if(index<=arr.length)
    {
       flag=1;
       printf("%d is found at %d\n",arr.A[index],index);

    }
    if(flag==0)
    {
        printf("No element found");
    }
}
void search(struct array arr,int key)
{

    int i,flag=0;
    for(i=0;i<arr.length;i++)
    {

        if(key==arr.A[i])
        {
           flag=1;
           printf("key is fount at index %d\n",i);

        }

    }
    if(flag==0)
    {

        printf("Not fount\n");
    }
}
void binsearch(struct array arr,int key)
{
    int start=0,end=arr.length-1,flag=0,mid;

    while(start<=end)
{      mid=(start+end)/2;

        if(arr.A[mid]==key)
        {
            flag=mid;
            break;
        }
        else if(key>arr.A[mid])
        {
            start=mid+1;
        }
        else if(key<arr.A[mid])
        {

            end=mid-1;
        }

}
if(flag==0)
{

    printf("no element found");
}
else
{

    printf("%d is found at index %d",key,flag);
}
}
int main()
{

    struct array arr;
    scanf("%d",&arr.n);
    arr.length=0;
    arr.A=(int*)malloc(sizeof(int)*arr.n); // (allocate memory in heap  wide space)
    append(&arr,10);
    append(&arr,20);
    append(&arr,30);
    append(&arr,40);
    insert(&arr,2,38);
    remmove(&arr,2);
    set(&arr,2,34);
    get(arr,6);
    search(arr,20);
    binsearch(arr,22);
    display(arr);


}
