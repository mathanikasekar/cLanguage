#include<stdio.h>
int main()
{
  int testcases,itr;
  scanf("%d",&testcases);
  for(itr=1;itr<=testcases;itr++)
  {
    char str[401];
    int reqdAscii,index,count[27]={0};
    scanf("\n%s",str);
    for(index=0;str[index]!=NULL;index++)
    {
      reqdAscii = str[index]-96;
      count[reqdAscii]++;
      if(count[reqdAscii]>reqdAscii)
        //definitely not a super Ascii string
        break;
    }
    if(str[index]!=NULL)//exit because of break;
      printf("Not super ASCII String\n");
    else//without breaking the loop
    {
      for(index=1;index<=26;index++)//condition fails
      {
        if(count[index]!=0 && count[index] != index)
          //not a super ascii string
          break;
      }
      if(index>26)
        printf("Super ASCII String\n");
      else
        printf("Not a Super ASCII String\n");
    }
  }
  return 0;
}
