#include<stdio.h>
int main()
{
  long long int n,factor;
  int k,count;
  //scanf("%llu,%d",&n,&k);
  n=9999999999;
  k=48;
  count=0;
  for(factor=n;factor>=1;factor--)
  {
    if(n%factor==0)//found a factor
    {
      count++;
      if(count==k)
      {
        printf("%d",factor);
        break;
      }
    }
  }
  //count=>number of factors
  //k=>input value
  if(k>count)
    printf("1");
  return 0;
}
