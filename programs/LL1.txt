#include<stdio.h>
#include<stdlib.h>
struct Node
{
int data;
struct Node *next;
};
struct Node* insertNode(struct Node* temp,int element,int pos)
{
  struct Node *p=temp;
  for(int i=0;i<pos-1;i++)
  {

      p=p->next;
  }
  struct Node* newNode =(struct Node*)malloc(sizeof(struct Node));
  newNode->data=element;
  if(pos==0)
  {
     newNode->next=temp;
     temp=newNode;

  }
  else{

    newNode->next=p->next;
    p->next=newNode;
  }
return temp;
}
struct Node* Lsearch(struct Node *temp,int key)//f(n)=n+1+1+n+n+1=3n+3=O(n)
{
    while(temp!=NULL)   //n+1 unit
    {
        if(temp->data==key)  // n unit
        return temp;        // 1 unit
        temp=temp->next;   //n unit

    }
    return NULL;   //1 unit
}
struct Node* Rsearch(struct Node *temp,int key)
{
    if(temp==NULL)
    return NULL;
    if(temp->data==key)
    return temp;
    Rsearch(temp->next,key);
}
void display(struct Node *head) //f(n)=n+n+n=3n->O(n)
{
    while(head!=NULL) //n unit
    {
        printf("%d ",head->data); //n unit
        head=head->next;  // n unit
    }
}
