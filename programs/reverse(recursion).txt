#include<stdio.h>
int power(int num)
{
    static  int ans=1;
    int res=0;
    if(num>0)
    {
        ans=10*power(num/10);
    }
    else
        return 1;
    res=ans;
    ans=1;
    return res;
}
int rev(int num)
{
    int pv;
    static ans=0;
    pv=power(num)/10;
    if(num>0)
    {
      ans=num%10*pv+rev(num/10);

    }
    else
        return 0;

    return ans;




}
