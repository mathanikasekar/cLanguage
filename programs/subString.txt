int substring(char *str1,char *str2)
{
    int i,j,k;
    for(i=0;str1[i]!='\0';i++)
    {
        if(str1[i]==str2[0])
        {
            for(j=0,k=i;
            str1[k]==str2[j]&&str2[j]!='\0'
            &&str1[k]!='\0';j++,k++);

}
