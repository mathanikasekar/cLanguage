struct Node
{
int data;
struct Node *next;
};
struct Node* mergell(struct Node *first,struct Node *second)//f(n)=6n+12=O(n)
                                                            //O(1)
{
 struct Node *last,*third;       // 2 unit 
 if(first->data<second->data)   // 1 unit
 {
     last=third=first;           // 1 unit
     first=first->next;          // 1 unit
     last->next=NULL;            // 1 unit
 }
 else{
    last=third=second;
    second=second->next;
    last->next=NULL;
 }
 while(first!=NULL&&second!=NULL)   //n+1 unit
 {
     if(first->data<second->data)   // n unit
     {
         last->next=first;        // n unit
         last=first;               // n unit
         first=first->next;       // n unit
         last->next=NULL;        // n unit
     }
     else{
        last->next=second;
        last=second;
        second=second->next;
        last->next=NULL;
     }
 }
 if(first!=NULL)       // 1 unit
    last->next=first;   // 1 unit
 if(second!=NULL)      // 1 unit
    last->next=second;   // 1 unit
 return third;     // 1 unit

}







int isLoop(struct Node *first)//f(n)=4n+4=O(n)
                              // 2 unit=2*n^0=O(1)
{
    struct Node *p,*q;   // 2 unit
    p=q=first;   // 1 unit
    do
    {

        q=q->next;  // n unit
        p=p->next;   //n unit
        p=p!=NULL?p->next:NULL;  // n unit
    }while(q!=NULL&&p!=NULL&&p!=q);  //n unit

    return p==q?1:0;   // 1 unit
}
