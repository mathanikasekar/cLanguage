#include <stdio.h>
#include <string.h>
#include <math.h>
#include <stdlib.h>

int main()
{

    long n,t,rem,f[10],i;
    scanf("%ld",&n);
    t=n;
    
    for(i=0;i<10;i++)
    {
        f[i]=0;
    }
    
    while(n!=0)
    {
        rem=n%10;
        f[rem]++;
        n=n/10;
    }
    
    for(i=0;i<10;i++)
    {
        printf("Frequency of %ld = %ld\n",i,f[i]);
    }
    
    return 0;
}
