power of 2 and multiple of 2 are not same
power of 2:
 2^0=1  2^1=2  2^2=4  2^3=8  2^4=16.......
multiple of 2:
2*0=0   2*1=2  2*2=4  2*3=6  2*4=8.....
#include <stdio.h>
int main()
{
    long n;
    printf("Enter a number :");
    scanf("%ld",&n);
    while(n%2==0)
    {
            n=n/2;
    }
    if(n==1)
             printf("YES\n");
    else
           printf("NO\n");
    return 0;
}
